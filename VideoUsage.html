Video Usage example Initiator of the video will have to be the server and client.
The rest of the user will be the clients.

Setting up Server function initServer() {
//server creation code
var server = new RTCStarServer();

//setting up widget necessary
var videoWidgetServer = new VideoWidgetServer(server);

//starting the server
server.start();

server.onServerEvent('Open',function(server_id){
$('#created_id').text('Server Id = '+server_id);
initClient(server_id);
});
}

Setting up Client function initClient(server_id) {
//server creation code
var client = new RTCStarClient();

//setting up widget necessary
var videoWidgetClient = new VideoWidgetClient(client);

//starting the client
client.start();

videoWidgetClient.onStart(function(localStream){
//set where you want to display your stream
$('#my-video').prop('src', URL.createObjectURL(localStream));
})
videoWidgetClient.onCall(function(stream){
//do what you want with the other person here!
$('#their-video').prop('src', URL.createObjectURL(stream));
})
}
